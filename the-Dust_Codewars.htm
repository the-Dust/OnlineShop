using System;
using System.Linq;
using System.Collections.Generic;

public class Brace 
{
    public static bool validBraces(String braces) 
    {
      string openBraces = "([{";
      Stack<char> stack = new Stack<char>();
      for(int i = 0; i<braces.Length; i++)
      {
        if(openBraces.Contains(braces[i]))
        {
          stack.Push(braces[i]);
        }
        else
        {
          //if open and close braces are complementary, they should be in the ASCII table beside each other
          // => Math.Abs(braces[i]-stack.Pop())>2
          if(stack.Count==0||Math.Abs(braces[i]-stack.Pop())>2)
          return false;
        }
      }
      return stack.Count==0;
    }
}

  {
    if(n<2)
    return 0;
    double count = 0;
    for(int i = n-1, j = 1; i>0; i--, j++)
    count+=i*(Math.Pow(9,j)*Fact(n-1))/(Fact(j-1)*Fact(i));
    return (long)count+CountZeroes(n-1);
  }

    res*=i;

  }

  {
      public BigInteger Get(int number)
      {
          if(number<3) return number;
          Matrix m = new Matrix{a11 = 2, a12 = 1, a21 = 1, a22 = 0};
          Matrix n = m;
          for (int i = 0; i < number-2; i++)

    public static Matrix operator*(Matrix lhs, Matrix rhs)
    {

             a12 = lhs.a11*rhs.a12 + lhs.a12*rhs.a22,
             a21 = lhs.a21*rhs.a11 + lhs.a22*rhs.a21,

  public static int HowManyBees(char[][] hive)
  {

    int count = 0;
    int w = hive[0].Length;
    int h = hive.Length;
    char[][] rotHive = new char[w][];
    for (int k = 0; k < w; k++)
    {
      rotHive[k] = new char[h];
      for (int i = 0; i < h; i++)

    }

    foreach(string str in arr)
    {
      count+=Regex.Matches(str, "bee").Count;
      count+=Regex.Matches(str, "eeb").Count;
    }
    return count;
  }

            if (today.Day < birthday.Day)

            if (months < 0)

using System;
using System.Linq;

public class Converter
{
  public static string ToBinary(int n)
  {
    //return Convert.ToString(n,2); //This works fine too :)
    if(n==0)
    return "0";
    string s = "";
    bool f = false; //flag of negative number
    if(n<0)
    {
      f = true;
      n*=-1;
    }
    while(n>0)
    {
      if(n%2==0)
      s="0"+s;
      else
      s="1"+s;
      n/=2;
    }
    if(f) //conversion to additional code
    //This is Two's complement code. First we invert the number, then adding 1 
    //to this number. And then we padding 1s to the left to 32 bit length
    {
      char[] ch = s.Select(x=>x=='1'?'0':'1').ToArray();
      for(int i = ch.Length-1; i>=0; i--)
      {
        if(ch[i]=='1') //Adding 1 in the binary system
        {
          ch.SetValue('0',i);
        }
        else
        {
          ch.SetValue('1', i);
          s = new string('1', 32 - ch.Length) + new string(ch);
          break;
        }
      }
    }
    return s;
  }
}

        long[] target = new long[arr.Count];

        {
